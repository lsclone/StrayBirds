* 1、文件读写

```
FILE* fp = fopen("1.ini", "r");
if (fp == nullptr) {
	return;
}
int x, y, nums = 0;
char lines[256];
while (feof(fp) == 0) {
	nums++;
	fgets(lines, 256, fp);
	sscanf(lines, "%d %d\n", &x, &y);
}
fclose(fp);

fp = fopen("1.ini", "w");
if (fp == nullptr) {
	return;
}
for (int i = 0; i < nums; i++) {
	fprintf(fp, "%d %d\n", x, y);
}
fclose(fp);
```
